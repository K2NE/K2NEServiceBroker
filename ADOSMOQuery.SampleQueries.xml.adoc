  <query name="Generate rundom numbers.">
    <command>
      SELECT
       randomNumber,
       COUNT(1) countOfRandomNumber
      FROM (
        SELECT
          ABS(CAST(NEWID() AS binary(6)) % 1000) + 1 randomNumber
        FROM sysobjects) sample
      GROUP BY randomNumber
      ORDER BY randomNumber
    </command>
